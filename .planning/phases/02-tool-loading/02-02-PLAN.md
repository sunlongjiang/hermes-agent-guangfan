    **Step 2: 在 evolution/tools/tool_loader.py 中实现 write_back_description():**

    ```python
    def write_back_description(
        file_path: Path,
        tool: ToolDescription,
        new_description: str,
        param_name: Optional[str] = None,
    ) -> None:
        """Replace a description in the source file, preserving format.

        Args:
            file_path: Path to the tool .py file
            tool: ToolDescription from extract_tool_descriptions()
            new_description: The evolved description text
            param_name: If set, replace this param's description instead of top-level
        """
    ```

    实现逻辑（per D-04, D-05, D-06）:

    1. `source = file_path.read_text()`
    2. 确定目标: 如果 param_name 指定，找到对应 ToolParam 的 desc_format 和原始 description；否则用 tool 的 top-level desc_format 和 description
    3. 在源码中定位原始描述的精确位置:
       - 先定位 schema 变量（`tool.schema_var_name`）
       - 在 schema 范围内定位 `"description":` 关键字
       - 如果是 param，先定位到 `"properties"` 块内的 `"param_name":` 子块，再在其中找 `"description":`
    4. 根据 desc_format 构建替换字符串:
       - SINGLE_LINE: `"new_description"` (双引号转义: `new_description.replace('"', '\\"')`)
       - PAREN_CONCAT: 将 new_description 按 ~70 字符拆行，用 `("line1 " "line2 ")` 格式包装。如果描述短（<= 80 字符），直接用单行 `"text"`
       - TRIPLE_QUOTE: `"""new_description"""`
       - VARIABLE_REF: 定位变量定义处，替换变量的值（如果原变量值是 paren_concat 格式，写回也用 paren_concat）
    5. 做字符串替换: `source = source[:start] + replacement + source[end:]`
    6. 写回文件: `file_path.write_text(source)`

    关键细节:
    - 转义处理: 对 SINGLE_LINE 格式，new_description 中的 `"` 替换为 `\"`，`\` 替换为 `\\`
    - 对 TRIPLE_QUOTE 格式，new_description 中的 `"""` 替换为 `\"""`
    - 定位使用 re.search 而非简单 str.find，因为 `"description"` 可能出现多次
    - 对 param_name 定位: 在 schema 变量范围内，先找 `"properties"` 块，再在其中找 `"param_name"` 子块，最后找其中的 `"description":` — 这确保不会误替换其他位置的 description

    @pytest.mark.skipif(not HERMES_AVAILABLE, reason="hermes-agent not available")
    class TestRealHermesAgent:
        # ... 已有的提取测试 ...

        def test_write_back_memory_tool(self, tmp_path):
            """Write back to a copy of memory_tool.py, verify round-trip."""
            path = get_hermes_agent_path()
            src = path / "tools" / "memory_tool.py"
            dst = tmp_path / "memory_tool.py"
            shutil.copy(src, dst)

            tools = extract_tool_descriptions(dst)
            memory = next(t for t in tools if t.name == "memory")
            original_params = [p.to_dict() for p in memory.params]

            new_desc = "EVOLVED: " + memory.description[:50]
            write_back_description(dst, memory, new_desc)

            # Round-trip 验证
            tools_after = extract_tool_descriptions(dst)
            memory_after = next(t for t in tools_after if t.name == "memory")
            assert memory_after.description == new_desc

            # Schema 结构不变
            after_params = [p.to_dict() for p in memory_after.params]
            for bp, ap in zip(original_params, after_params):
                assert bp["name"] == ap["name"]
                assert bp["type"] == ap["type"]
                assert bp["required"] == ap["required"]

            # 语法检查
            py_compile.compile(str(dst), doraise=True)

        def test_write_back_all_tools_no_crash(self, tmp_path):
            """Write back to copies of all tool files without errors."""
            path = get_hermes_agent_path()
            files = discover_tool_files(path)
            success_count = 0
            for src in files:
                dst = tmp_path / src.name
                shutil.copy(src, dst)
                tools = extract_tool_descriptions(dst)
                for tool in tools:
                    new_desc = "EVOLVED: " + tool.description[:30]
                    write_back_description(dst, tool, new_desc)
                    # 验证语法
                    py_compile.compile(str(dst), doraise=True)
                    success_count += 1
            assert success_count >= 30
